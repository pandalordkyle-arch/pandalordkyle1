<script>
/* ===== PROFILE ===== */
let myId = localStorage.getItem("playerId");
if (!myId) {
    myId = Math.random().toString(36).substr(2,9);
    localStorage.setItem("playerId", myId);
}

let profile = JSON.parse(localStorage.getItem("profile") || "null");
if (!profile) {
    profile = { name:"Player", avatar:"🙂" };
    localStorage.setItem("profile", JSON.stringify(profile));
}

function refreshProfile() {
    document.getElementById("profileName").innerText = profile.name;
    document.getElementById("profileAvatar").innerText = profile.avatar;
}

function changeName() {
    let name = prompt("Enter new name:");
    if (!name) return;
    profile.name = name;
    localStorage.setItem("profile", JSON.stringify(profile));
    refreshProfile();
}

function setAvatar(emoji) {
    profile.avatar = emoji;
    localStorage.setItem("profile", JSON.stringify(profile));
    refreshProfile();
}

window.onload = refreshProfile;

/* ===== THEME ===== */
let theme = {
    background: 0x000000,
    ground1: 0x111111,
    ground2: 0x222222,
    player: 0x00ffcc,
    block: 0xff3333
};

function setTheme(name) {
    if (name === "neon") {
        theme = {
            background: 0x000000,
            ground1: 0x111111,
            ground2: 0x00ffcc,
            player: 0x00ffcc,
            block: 0xff00ff
        };
    }

    if (name === "dark") {
        theme = {
            background: 0x050505,
            ground1: 0x101010,
            ground2: 0x202020,
            player: 0xffffff,
            block: 0xff3333
        };
    }

    if (name === "fire") {
        theme = {
            background: 0x000000,
            ground1: 0x220000,
            ground2: 0x440000,
            player: 0xff6600,
            block: 0xff0000
        };
    }

    if (scene) scene.background = new THREE.Color(theme.background);
}

/* ===== ROOM & CHAT ===== */
let rooms = {};
let roomId = null;

function sendChat() {
    const input = document.getElementById("chatInput");
    const msg = input.value.trim();
    if (!msg || !roomId) return;
    input.value = "";

    rooms[roomId].chat.push(`${profile.avatar} ${profile.name}: ${msg}`);
    renderChat();
}

function renderChat() {
    const box = document.getElementById("chatBox");
    box.innerHTML = rooms[roomId].chat.map(m => `<div>${m}</div>`).join("");
    box.scrollTop = box.scrollHeight;
}

window.joinRoom = function() {
    roomId = prompt("Room name:");
    if (!roomId) return;

    if (!rooms[roomId]) rooms[roomId] = { players:{}, chat:[] };

    rooms[roomId].players[myId] = {
        name: profile.name,
        avatar: profile.avatar,
        score: 0,
        health: 100,
        level: 1
    };

    document.getElementById("menu").style.display = "none";
    document.getElementById("hud").style.display = "block";
    document.getElementById("chat").style.display = "block";
    document.getElementById("roomId").innerText = roomId;

    listenChat();
    init();
    animate();
};

/* ===== LEADERBOARD ===== */
function saveScore() {
    let board = JSON.parse(localStorage.getItem("board") || "[]");

    board.push({
        name: `${profile.avatar} ${profile.name}`,
        score
    });

    board.sort((a,b) => b.score - a.score);
    localStorage.setItem("board", JSON.stringify(board.slice(0,10)));
}

window.showLeaderboard = function() {
    document.getElementById("menu").style.display = "none";
    document.getElementById("leaderboard").style.display = "block";

    let board = JSON.parse(localStorage.getItem("board") || "[]");
    let list = document.getElementById("scores");
    list.innerHTML = board.map(b => `<li>${b.name} - ${b.score}</li>`).join("");
};

window.backToMenu = function() {
    document.getElementById("leaderboard").style.display = "none";
    document.getElementById("menu").style.display = "block";
};

/* ===== GAME ===== */
let scene, camera, renderer, player;
let health = 100;
let score = 0;
let level = 1;
let wave = 1;
let blocks = [];
let particles = [];
let groundTiles = [];
let gameRunning = false;

let blockSpeed = 0.04;
let spawnRate = 0.015;

let playZone = 4;
let minZone = 2.5;

let boss = null;
let bossHealth = 0;

/* GROUND */
function buildGround() {
    for (let x = -5; x <= 5; x++) {
        for (let z = -5; z <= 5; z++) {
            let geo = new THREE.PlaneGeometry(1, 1);
            let mat = new THREE.MeshStandardMaterial({
                color: (x + z) % 2 ? theme.ground1 : theme.ground2,
                roughness: 0.9
            });

            let tile = new THREE.Mesh(geo, mat);
            tile.rotation.x = -Math.PI / 2;
            tile.position.set(x, -3.2, z);

            scene.add(tile);
            groundTiles.push(tile);
        }
    }
}

/* BOUNDARIES */
function buildBoundaries() {
    let geo = new THREE.BoxGeometry(0.2, 3, 10);
    let mat = new THREE.MeshStandardMaterial({ color:0x444444 });

    let left = new THREE.Mesh(geo, mat);
    left.position.set(-playZone - 0.2, -2, 0);
    scene.add(left);

    let right = new THREE.Mesh(geo, mat);
    right.position.set(playZone + 0.2, -2, 0);
    scene.add(right);
}

/* INIT */
function init() {
    scene = new THREE.Scene();
    scene.background = new THREE.Color(theme.background);

    camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
    camera.position.z = 7;

    renderer = new THREE.WebGLRenderer({ antialias:true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setPixelRatio(window.devicePixelRatio);
    document.body.appendChild(renderer.domElement);

    let light = new THREE.DirectionalLight(0xffffff, 1);
    light.position.set(5, 10, 5);
    scene.add(light);

    let ambient = new THREE.AmbientLight(0x404040);
    scene.add(ambient);

    buildGround();
    buildBoundaries();

    let bodyGeo = new THREE.CylinderGeometry(0.6, 0.6, 2, 12);
    let bodyMat = new THREE.MeshStandardMaterial({ color: theme.player });
    player = new THREE.Mesh(bodyGeo, bodyMat);

    let headGeo = new THREE.SphereGeometry(0.6, 12, 12);
    let head = new THREE.Mesh(headGeo, bodyMat);
    head.position.y = 1.4;
    player.add(head);

    player.position.y = -2.5;
    scene.add(player);

    document.addEventListener("keydown", movePlayer);
    window.addEventListener("resize", onResize);

    gameRunning = true;
}

/* WAVES */
function nextWave() {
    wave++;
    level++;

    blockSpeed += 0.01;
    spawnRate += 0.005;

    playZone -= 0.2;
    if (playZone < minZone) playZone = minZone;

    buildBoundaries();

    if (wave % 5 === 0) spawnBoss();

    if (spawnRate > 0.12) spawnRate = 0.12;
    if (blockSpeed > 0.25) blockSpeed = 0.25;
}

/* BOSS */
function spawnBoss() {
    let geo = new THREE.BoxGeometry(3,3,3);
    let mat = new THREE.MeshStandardMaterial({ color: 0xff0000 });

    boss = new THREE.Mesh(geo, mat);
    boss.position.set(0, 8, 0);

    bossHealth = 50 + (wave * 10);
    scene.add(boss);
}

/* RESIZE */
function onResize() {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
}

/* MOVE */
function movePlayer(e) {
    if (!player) return;

    if (e.key.toLowerCase() === "a") player.position.x -= 0.6;
    if (e.key.toLowerCase() === "d") player.position.x += 0.6;

    if (player.position.x < -playZone) player.position.x = -playZone + 0.4;
    if (player.position.x > playZone) player.position.x = playZone - 0.4;
}

/* BLOCKS */
function spawnBlock() {
    let geo = new THREE.BoxGeometry(1, 1, 1);
    let mat = new THREE.MeshStandardMaterial({
        color: theme.block,
        emissive: 0x330000,
        roughness: 0.4,
        metalness: 0.2
    });

    let block = new THREE.Mesh(geo, mat);

    block.position.x = (Math.random() * (playZone * 2)) - playZone;
    block.position.y = 8;
    block.position.z = 0;

    scene.add(block);
    blocks.push(block);
}

/* PARTICLES */
function explode(x, y) {
    for (let i = 0; i < 15; i++) {
        let geo = new THREE.SphereGeometry(0.1);
        let mat = new THREE.MeshStandardMaterial({ color:0xffff00 });
        let p = new THREE.Mesh(geo, mat);

        p.position.set(x, y, 0);
        scene.add(p);

        particles.push({
            mesh: p,
            vx: (Math.random()-0.5) * 0.2,
            vy: Math.random() * 0.2
        });
    }
}

/* LOOP */
function animate() {
    if (!gameRunning) return;
    requestAnimationFrame(animate);

    if (Math.random() < spawnRate && blocks.length < 20) spawnBlock();

    blocks.forEach((block, i) => {
        block.position.y -= blockSpeed * 0.9;

        if (block.position.y < -4) {
            scene.remove(block);
            blocks.splice(i, 1);
            score++;
        }

        if (Math.abs(block.position.x - player.position.x) < 1 &&
            Math.abs(block.position.y - player.position.y) < 1) {

            health -= 10;
            explode(block.position.x, block.position.y);
            scene.remove(block);
            blocks.splice(i, 1);

            if (health <= 0) gameOver();
        }

        if (boss && Math.abs(block.position.x - boss.position.x) < 2) {
            bossHealth -= 5;
            if (bossHealth <= 0) {
                scene.remove(boss);
                boss = null;
                score += 100;
            }
        }
    });

    particles.forEach((p, i) => {
        p.mesh.position.x += p.vx;
        p.mesh.position.y += p.vy;
        p.vy -= 0.01;

        if (p.mesh.position.y < -5) {
            scene.remove(p.mesh);
            particles.splice(i,1);
        }
    });

    if (boss) {
        boss.position.y -= blockSpeed * 0.2;
        if (boss.position.y < -4) {
            scene.remove(boss);
            boss = null;
            score += 50;
        }
    }

    if (score > level * 10) nextWave();

    if (player.position.x < -playZone * 1.3 || player.position.x > playZone * 1.3) {
        player.position.x = 0;
        health -= 5;
    }

    document.getElementById("health").innerText = health;
    document.getElementById("score").innerText = score;
    document.getElementById("level").innerText = level;
    document.getElementById("wave").innerText = wave;

    renderer.render(scene, camera);
}

/* GAME OVER */
function gameOver() {
    gameRunning = false;
    saveScore();
    alert("Game Over!");
    location.reload();
}

/* CHAT */
function listenChat() {
    renderChat();
}
